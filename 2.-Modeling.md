``` r
get_robust_se <- function(lm_model)  {
  matrix <- vcovHC(lm_model, type = "HC0")
  robust_se <- sqrt(diag(matrix))
  return(robust_se)
}
```

``` r
get_r2 <- function(model){  
 SSe <- sum((model$resid)^2)
 observed <- model$resid + model$fitted
 SSt <- sum((observed - mean(observed))^2)
 value <- 1 - SSe/SSt
 return(value)
}
```
